SELECT * FROM Worker WHERE FIRST_NAME = 'Aditi' OR FIRST_NAME = 'Vivek';

# IN operator
select worker_id, first_name, department, salary from worker 
where worker_id in (2 ,3 ,4);

select worker_id, first_name, department, salary from worker 
where worker_id not in (1 ,3 ,4);

# BETWEEN operator
select * from worker where salary between 110000 and 300000;
select * from worker where salary not between 100000 and 200000;

select * from worker where salary between 100000 and 500000
and department not in ('hr','admin');

select * from worker where salary between 100000 and 600000
and department in ('admin');

# LIKE operator
select * from worker where first_name like 'A%';
select * from worker where first_name like '__t%';
select * from worker where first_name like '__h__i%';

# Aggregate functions
select min(salary) from worker;
select min(salary) from worker where department= 'admin';
select max(salary) from worker;
select max(salary) from worker where department = 'hr';
select count(worker_id) from worker;
select sum(salary) from worker where department = 'account';
select avg(salary) from worker;

select min(salary) from worker 
where department = 'admin' and first_name not like '%a%';

# Union operator
show tables from vitcampus1;

select first_name from worker union all select first_name from worker;

select worker_id as Rollnumber from worker;
select distinct(department) from worker;

# Case statement
select worker_id, first_name,
Case
  when salary >400000 then 'Good salary'
  when salary >= 100000 and salary < 299999 then 'average salary'
  else 'poor'
End 
AS Status
from worker;
