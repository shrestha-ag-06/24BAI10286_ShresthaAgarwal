# JOIN statement
use vitcampus1;

create table departments(
did int primary key,
dname varchar(25) not null unique
);
insert into departments values
(101,'CSE'),(102,'ECE'),(103,'CIVIL'),(106,'AERO'),(107,'MECH');

select * from departments order by did;

create table HOD(
Hid int primary key,
hname varchar (20) not null unique);
insert into HOD values
(101,'Shreya'),(102,'Arun'),(104,'Karthik'),(105,'Ananya');

select * from hod order by hid;

select * from departments cross join hod order by did asc;
select * from departments inner join hod where departments.did = hod.hid;
select * from departments natural join hod;
select * from departments left outer join hod on (departments.did = hod.hid) order by did asc;
select * from departments right outer join hod on (departments.did = hod.hid) order by did asc;
